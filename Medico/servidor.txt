from flask import Flask, request
from main import enviarDados

app = Flask("Servidor")

@app.route("/olamundo", methods=["GET"])
def olaMundo():
    return {"ola": "mundo"}

@app.route("/Sintomas", methods=["POST"])
def enviarDados():

    body = request.get_json()
    
    if("nome" not in body):
        return geraResponse(400, "O parametro nome é obrigatorio")
    
    if("valor" not in body):
        return geraResponse(400, "O parametro valor é obrigatorio")

    sintomas = enviarDados(body["nome"], body["valor"])

    return geraResponse(200, "sintomas enviado", "user" , sintomas)

def geraResponse(status, mensagem, nome_do_conteudo=False, conteudo=False):
    response = {}
    response["status"] = status
    response["mensagem"] = mensagem

    if(nome_do_conteudo and conteudo):
        response[nome_do_conteudo] = conteudo
    
    return response

app.run()
